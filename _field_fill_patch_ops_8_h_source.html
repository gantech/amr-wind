133  amrex::Vector<amrex::MultiFab*> get_mfab_vec(int lev)
134  {
135  const int nstates = amrex::min(m_field.num_time_states(), 2);
136  amrex::Vector<amrex::MultiFab*> ret;
137 
138  // The states in the FieldInfo data are ordered from newest to oldest,
139  // so swap the order
140  for (int i = nstates - 1; i >= 0; --i) {
141  const auto fstate = static_cast<FieldState>(i);
142  ret.push_back(&m_field.state(fstate)(lev));
143  }
144  return ret;
145  }
